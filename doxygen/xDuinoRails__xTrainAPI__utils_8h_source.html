15// Helper to convert hex string to bytes
16static std::vector<uint8_t> hexToBytes(const std::string& hex) {
17 std::vector<uint8_t> bytes;
18 for (size_t i = 0; i < hex.length(); i += 2) {
19 std::string byteString = hex.substr(i, 2);
20 char byte = (char)strtol(byteString.c_str(), NULL, 16);
21 bytes.push_back(byte);
22 }
23 return bytes;
24}

1085 String getParamValue(const String& params, const char* key) {
1086 String searchKey = String(key) + "=\"";
1087 int keyIndex = params.indexOf(searchKey);
1088 if (keyIndex == -1) {
1089 return "";
1090 }
1091 int valueStartIndex = keyIndex + searchKey.length();
1092 int valueEndIndex = params.indexOf('"', valueStartIndex);
1093 if (valueEndIndex == -1) {
1094 return "";
1095 }
1096 return params.substring(valueStartIndex, valueEndIndex);
1097 }
1098
1099 void parse(const String& command) {
1100 if (command.length() == 0) {
1101 return;
1102 }
1103
1104 // Try parsing extended format first
1105 int firstSpace = command.indexOf(' ');
1106 String cmd;
1107 String params;
1108 if (firstSpace != -1) {
1109 cmd = command.substring(0, firstSpace);
1110 params = command.substring(firstSpace + 1);
1111 } else {
1112 cmd = command;
1113 }
1114
1115 if (cmd == "POWER") {
1116 String state = getParamValue(params, "state");
1117 if (state == "ON") {
1118 _listener->onTrackPowerChange(PowerState::ON);
1119 } else if (state == "OFF") {
1120 _listener->onTrackPowerChange(PowerState::OFF);
1121 }
1122 return;
1123 } else if (cmd == "THROTTLE") {
1124 uint16_t cab = getParamValue(params, "cab").toInt();
1125 int speed = getParamValue(params, "speed").toInt();
1126 int dir = getParamValue(params, "direction").toInt();
1127 int steps = getParamValue(params, "steps").toInt();
1128 if (steps == 0) steps = 128; // default
1129
1130 LocoHandle loco = {cab, Protocol::DCC, 0};
1131 _listener->onLocoSpeedChange(loco, speed, (Direction)dir, steps);
1132 return;
1133 } else if (cmd == "FUNCTION") {
1134 uint16_t cab = getParamValue(params, "cab").toInt();
1135 int func = getParamValue(params, "function").toInt();
1136 int state = getParamValue(params, "state").toInt();
1137
1138 LocoHandle loco = {cab, Protocol::DCC, 0};
1139 _listener->onLocoFunctionChange(loco, func, state);
1140 return;
1141 } else if (cmd == "TURNOUT") {
