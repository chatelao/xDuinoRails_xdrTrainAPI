664class XmlParser {
665public:
666 XmlParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {
667 yxml_init(&_xml, _stack, sizeof(_stack));
668 }
669
670 void parse(const String& xml) {
671 LocoHandle loco;
672 float speedPercent;
673 Direction direction;
674 int speedSteps;
675
676 for (size_t i = 0; i < xml.length(); i++) {
677 yxml_ret_t r = yxml_parse(&_xml, xml[i]);
678 if (r == YXML_ELEMSTART && strcmp(_xml.elem, "loco") == 0) {
679 // do nothing
680 } else if (r == YXML_ATTRSTART && strcmp(_xml.attr, "address") == 0) {
681 // do nothing
682 } else if (r == YXML_ATTRVAL) {
683 if (strcmp(_xml.elem, "loco") == 0 && strcmp(_xml.attr, "address") == 0) {
684 loco.address = atoi(_xml.data);
685 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedPercent") == 0) {
686 speedPercent = atof(_xml.data);
687 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "direction") == 0) {
688 direction = strcmp(_xml.data, "FORWARD") == 0 ? Direction::FORWARD : Direction::REVERSE;
689 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedSteps") == 0) {
690 speedSteps = atoi(_xml.data);
691 }
692 } else if (r == YXML_ELEMEND && strcmp(_xml.elem, "event") == 0) {
693 _listener->onLocoSpeedChange(loco, speedPercent, direction, speedSteps);
694 }
695 }
696 yxml_eof(&_xml);
697 }
698
699private:
700 IUnifiedModelTrainListener* _listener;
701 yxml_t _xml;
702 uint8_t _stack[4096];
703};

705class CmdLineParser {
706public:
707 CmdLineParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {}
708
709 String getParamValue(const String& params, const char* key) {
710 String searchKey = String(key) + "=\"";
711 int keyIndex = params.indexOf(searchKey);
712 if (keyIndex == -1) {
713 return "";
714 }
715 int valueStartIndex = keyIndex + searchKey.length();
716 int valueEndIndex = params.indexOf('"', valueStartIndex);
717 if (valueEndIndex == -1) {
718 return "";
719 }
720 return params.substring(valueStartIndex, valueEndIndex);
721 }
722
723 void parse(const String& command) {
724 if (command.length() == 0) {
725 return;
726 }
727
728 // Try parsing extended format first
729 int firstSpace = command.indexOf(' ');
730 String cmd;
731 String params;
732 if (firstSpace != -1) {
733 cmd = command.substring(0, firstSpace);
734 params = command.substring(firstSpace + 1);
735 } else {
736 cmd = command;
737 }
738
739 if (cmd == "POWER") {
740 String state = getParamValue(params, "state");
741 if (state == "ON") {
742 _listener->onTrackPowerChange(PowerState::ON);
743 } else if (state == "OFF") {
744 _listener->onTrackPowerChange(PowerState::OFF);
745 }
746 return;
747 } else if (cmd == "THROTTLE") {
748 uint16_t cab = getParamValue(params, "cab").toInt();
749 int speed = getParamValue(params, "speed").toInt();
750 int dir = getParamValue(params, "direction").toInt();
751 int steps = getParamValue(params, "steps").toInt();
752 if (steps == 0) steps = 128; // default
753
754 LocoHandle loco = {cab, Protocol::DCC, 0};
755 _listener->onLocoSpeedChange(loco, speed, (Direction)dir, steps);
756 return;
757 } else if (cmd == "FUNCTION") {
758 uint16_t cab = getParamValue(params, "cab").toInt();
759 int func = getParamValue(params, "function").toInt();
760 int state = getParamValue(params, "state").toInt();
761
762 LocoHandle loco = {cab, Protocol::DCC, 0};
763 _listener->onLocoFunctionChange(loco, func, state);
764 return;
765 } else if (cmd == "TURNOUT") {
766 uint16_t addr = getParamValue(params, "id").toInt();
767 int state = getParamValue(params, "state").toInt();
768 _listener->onTurnoutChange(addr, state, false);
769 return;
770 } else if (cmd == "SIGNAL") {
771 uint16_t addr = getParamValue(params, "id").toInt();
772 int aspect = getParamValue(params, "aspect").toInt();
773 _listener->onSignalAspectChange(addr, aspect, false);
774 return;
775 } else if (cmd == "MECH_SYNC") {
776 uint16_t cab = getParamValue(params, "cab").toInt();
777 String typeStr = getParamValue(params, "type");
778 uint8_t value = getParamValue(params, "value").toInt();
779 SyncType type;
780 if (typeStr == "CAM_PULSE") type = SyncType::CAM_PULSE;
781 else if (typeStr == "CYLINDER_CYCLE") type = SyncType::CYLINDER_CYCLE;
782 else if (typeStr == "GEAR_CHANGE_UP") type = SyncType::GEAR_CHANGE_UP;
783 else if (typeStr == "GEAR_CHANGE_DOWN") type = SyncType::GEAR_CHANGE_DOWN;
784 else if (typeStr == "BRAKE_SQUEAL_START") type = SyncType::BRAKE_SQUEAL_START;
785 else if (typeStr == "DOOR_MOVEMENT") type = SyncType::DOOR_MOVEMENT;
786
787 LocoHandle loco = {cab, Protocol::DCC, 0};
788 _listener->onLocoEventSync(loco, type, value);
789 return;
790 }
791
792 // If not an extended command, try parsing legacy format
793 char cmdChar = command.charAt(0);
794 params = command.substring(1);
795 params.trim();
796
797 switch (cmdChar) {
798 case '1':
799 _listener->onTrackPowerChange(PowerState::ON);
800 break;
801 case '0':
802 _listener->onTrackPowerChange(PowerState::OFF);
803 break;
804 case 't':
805 {
806 int firstSpace = params.indexOf(' ');
807 int secondSpace = params.indexOf(' ', firstSpace + 1);
808 if (firstSpace != -1 && secondSpace != -1) {
809 uint16_t cab = params.substring(0, firstSpace).toInt();
810 int speed = params.substring(firstSpace + 1, secondSpace).toInt();
811 int dir = params.substring(secondSpace + 1).toInt();
812
813 LocoHandle loco = {cab, Protocol::DCC, 0};
814 _listener->onLocoSpeedChange(loco, speed, (Direction)dir, 128);
815 }
816 }
817 break;
818 case 'f':
819 {
820 int firstSpace = params.indexOf(' ');
821 int secondSpace = params.indexOf(' ', firstSpace + 1);
822 if (firstSpace != -1 && secondSpace != -1) {
823 uint16_t cab = params.substring(0, firstSpace).toInt();
824 int func = params.substring(firstSpace + 1, secondSpace).toInt();
825 int state = params.substring(secondSpace + 1).toInt();
826
827 LocoHandle loco = {cab, Protocol::DCC, 0};
828 _listener->onLocoFunctionChange(loco, func, state);
829 }
830 }
831 break;
832 case 'T':
833 {
834 int firstSpace = params.indexOf(' ');
835 if (firstSpace != -1) {
836 uint16_t addr = params.substring(0, firstSpace).toInt();
837 int state = params.substring(firstSpace + 1).toInt();
838 _listener->onTurnoutChange(addr, state, false);
839 }
840 }
841 break;
842 case 'S':
843 {
844 int firstSpace = params.indexOf(' ');
845 if (firstSpace != -1) {
846 uint16_t addr = params.substring(0, firstSpace).toInt();
847 int aspect = params.substring(firstSpace + 1).toInt();
848 _listener->onSignalAspectChange(addr, aspect, false);
849 }
850 }
851 break;
852 case 'm':
853 {
854 int firstSpace = params.indexOf(' ');
855 int secondSpace = params.indexOf(' ', firstSpace + 1);
856 if (firstSpace != -1 && secondSpace != -1) {
857 uint16_t cab = params.substring(0, firstSpace).toInt();
858 int type = params.substring(firstSpace + 1, secondSpace).toInt();
859 int value = params.substring(secondSpace + 1).toInt();
860
861 LocoHandle loco = {cab, Protocol::DCC, 0};
862 _listener->onLocoEventSync(loco, (SyncType)type, value);
863 }
864 }
865 break;
866 }
867 }
868
869private:
870 IUnifiedModelTrainListener* _listener;
871};
