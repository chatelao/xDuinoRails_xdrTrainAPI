380class XmlParser {
381public:
382 XmlParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {
383 yxml_init(&_xml, _stack, sizeof(_stack));
384 }
385
386 void parse(const String& xml) {
387 LocoHandle loco;
388 float speedPercent;
389 Direction direction;
390 int speedSteps;
391
392 for (size_t i = 0; i < xml.length(); i++) {
393 yxml_ret_t r = yxml_parse(&_xml, xml[i]);
394 if (r == YXML_ELEMSTART && strcmp(_xml.elem, "loco") == 0) {
395 // do nothing
396 } else if (r == YXML_ATTRSTART && strcmp(_xml.attr, "address") == 0) {
397 // do nothing
398 } else if (r == YXML_ATTRVAL) {
399 if (strcmp(_xml.elem, "loco") == 0 && strcmp(_xml.attr, "address") == 0) {
400 loco.address = atoi(_xml.data);
401 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedPercent") == 0) {
402 speedPercent = atof(_xml.data);
403 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "direction") == 0) {
404 direction = strcmp(_xml.data, "FORWARD") == 0 ? Direction::FORWARD : Direction::REVERSE;
405 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedSteps") == 0) {
406 speedSteps = atoi(_xml.data);
407 }
408 } else if (r == YXML_ELEMEND && strcmp(_xml.elem, "event") == 0) {
409 _listener->onLocoSpeedChange(loco, speedPercent, direction, speedSteps);
410 }
411 }
412 yxml_eof(&_xml);
413 }
414
415private:
416 IUnifiedModelTrainListener* _listener;
417 yxml_t _xml;
418 uint8_t _stack[4096];

422public:
423 CmdLineParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {}

425 String getParamValue(const String& params, const char* key) {
426 String searchKey = String(key) + "=\"";
427 int keyIndex = params.indexOf(searchKey);
428 if (keyIndex == -1) {
429 return "";
430 }
431 int valueStartIndex = keyIndex + searchKey.length();
432 int valueEndIndex = params.indexOf('"', valueStartIndex);
433 if (valueEndIndex == -1) {
434 return "";
435 }
436 return params.substring(valueStartIndex, valueEndIndex);
437 }
438
439 void parse(const String& command) {
440 if (command.length() == 0) {
441 return;
442 }
443
444 // Try parsing extended format first
445 int firstSpace = command.indexOf(' ');
446 String cmd;
447 String params;
448 if (firstSpace != -1) {
449 cmd = command.substring(0, firstSpace);
450 params = command.substring(firstSpace + 1);
451 } else {
452 cmd = command;
453 }
454
455 if (cmd == "POWER") {
456 String state = getParamValue(params, "state");
457 if (state == "ON") {
458 _listener->onTrackPowerChange(PowerState::ON);
459 } else if (state == "OFF") {
460 _listener->onTrackPowerChange(PowerState::OFF);
461 }
462 return;
463 } else if (cmd == "THROTTLE") {
464 uint16_t cab = getParamValue(params, "cab").toInt();
465 int speed = getParamValue(params, "speed").toInt();
466 int dir = getParamValue(params, "direction").toInt();
467 int steps = getParamValue(params, "steps").toInt();
468 if (steps == 0) steps = 128; // default
469
470 LocoHandle loco = {cab, Protocol::DCC, 0};
471 _listener->onLocoSpeedChange(loco, speed, (Direction)dir, steps);
472 return;
473 } else if (cmd == "FUNCTION") {
474 uint16_t cab = getParamValue(params, "cab").toInt();
475 int func = getParamValue(params, "function").toInt();
476 int state = getParamValue(params, "state").toInt();
477
478 LocoHandle loco = {cab, Protocol::DCC, 0};
479 _listener->onLocoFunctionChange(loco, func, state);
480 return;
481 } else if (cmd == "TURNOUT") {
482 uint16_t addr = getParamValue(params, "id").toInt();
483 int state = getParamValue(params, "state").toInt();
484 _listener->onTurnoutChange(addr, state, false);
485 return;
486 } else if (cmd == "SIGNAL") {
487 uint16_t addr = getParamValue(params, "id").toInt();
488 int aspect = getParamValue(params, "aspect").toInt();
489 _listener->onSignalAspectChange(addr, aspect, false);
490 return;
491 } else if (cmd == "MECH_SYNC") {
492 uint16_t cab = getParamValue(params, "cab").toInt();
493 String typeStr = getParamValue(params, "type");
494 uint8_t value = getParamValue(params, "value").toInt();
495 SyncType type;
496 if (typeStr == "CAM_PULSE") type = SyncType::CAM_PULSE;
497 else if (typeStr == "CYLINDER_CYCLE") type = SyncType::CYLINDER_CYCLE;
498 else if (typeStr == "GEAR_CHANGE_UP") type = SyncType::GEAR_CHANGE_UP;
499 else if (typeStr == "GEAR_CHANGE_DOWN") type = SyncType::GEAR_CHANGE_DOWN;
500 else if (typeStr == "BRAKE_SQUEAL_START") type = SyncType::BRAKE_SQUEAL_START;
501 else if (typeStr == "DOOR_MOVEMENT") type = SyncType::DOOR_MOVEMENT;
502
503 LocoHandle loco = {cab, Protocol::DCC, 0};
504 _listener->onLocoEventSync(loco, type, value);
505 return;
506 }
507
508 // If not an extended command, try parsing legacy format
509 char cmdChar = command.charAt(0);
510 params = command.substring(1);
511 params.trim();
512
513 switch (cmdChar) {
514 case '1':
515 _listener->onTrackPowerChange(PowerState::ON);
516 break;
517 case '0':
518 _listener->onTrackPowerChange(PowerState::OFF);
519 break;
520 case 't':
521 {
522 int firstSpace = params.indexOf(' ');
523 int secondSpace = params.indexOf(' ', firstSpace + 1);
524 if (firstSpace != -1 && secondSpace != -1) {
525 uint16_t cab = params.substring(0, firstSpace).toInt();
526 int speed = params.substring(firstSpace + 1, secondSpace).toInt();
527 int dir = params.substring(secondSpace + 1).toInt();
528
529 LocoHandle loco = {cab, Protocol::DCC, 0};
530 _listener->onLocoSpeedChange(loco, speed, (Direction)dir, 128);
531 }
532 }
533 break;
534 case 'f':
535 {
536 int firstSpace = params.indexOf(' ');
537 int secondSpace = params.indexOf(' ', firstSpace + 1);
538 if (firstSpace != -1 && secondSpace != -1) {
539 uint16_t cab = params.substring(0, firstSpace).toInt();
540 int func = params.substring(firstSpace + 1, secondSpace).toInt();
541 int state = params.substring(secondSpace + 1).toInt();
542
543 LocoHandle loco = {cab, Protocol::DCC, 0};
544 _listener->onLocoFunctionChange(loco, func, state);
545 }
546 }
547 break;
548 case 'T':
549 {
550 int firstSpace = params.indexOf(' ');
551 if (firstSpace != -1) {
552 uint16_t addr = params.substring(0, firstSpace).toInt();
553 int state = params.substring(firstSpace + 1).toInt();
554 _listener->onTurnoutChange(addr, state, false);
555 }
556 }
557 break;
558 case 'S':
559 {
560 int firstSpace = params.indexOf(' ');
561 if (firstSpace != -1) {
562 uint16_t addr = params.substring(0, firstSpace).toInt();
563 int aspect = params.substring(firstSpace + 1).toInt();
564 _listener->onSignalAspectChange(addr, aspect, false);
565 }
566 }
567 break;
568 case 'm':
569 {
570 int firstSpace = params.indexOf(' ');
571 int secondSpace = params.indexOf(' ', firstSpace + 1);
572 if (firstSpace != -1 && secondSpace != -1) {
573 uint16_t cab = params.substring(0, firstSpace).toInt();
574 int type = params.substring(firstSpace + 1, secondSpace).toInt();
575 int value = params.substring(secondSpace + 1).toInt();
576
577 LocoHandle loco = {cab, Protocol::DCC, 0};
578 _listener->onLocoEventSync(loco, (SyncType)type, value);
579 }
580 }
581 break;
582 }
583 }
