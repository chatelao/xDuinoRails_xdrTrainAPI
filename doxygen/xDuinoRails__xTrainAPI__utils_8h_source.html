235class XmlParser {
236public:
237 XmlParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {
238 yxml_init(&_xml, _stack, sizeof(_stack));
239 }
240
241 void parse(const String& xml) {
242 LocoHandle loco;
243 float speedPercent;
244 Direction direction;
245 int speedSteps;
246
247 for (size_t i = 0; i < xml.length(); i++) {
248 yxml_ret_t r = yxml_parse(&_xml, xml[i]);
249 if (r == YXML_ELEMSTART && strcmp(_xml.elem, "loco") == 0) {
250 // do nothing
251 } else if (r == YXML_ATTRSTART && strcmp(_xml.attr, "address") == 0) {
252 // do nothing
253 } else if (r == YXML_ATTRVAL) {
254 if (strcmp(_xml.elem, "loco") == 0 && strcmp(_xml.attr, "address") == 0) {
255 loco.address = atoi(_xml.data);
256 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedPercent") == 0) {
257 speedPercent = atof(_xml.data);
258 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "direction") == 0) {
259 direction = strcmp(_xml.data, "FORWARD") == 0 ? Direction::FORWARD : Direction::REVERSE;
260 } else if (strcmp(_xml.elem, "speed") == 0 && strcmp(_xml.attr, "speedSteps") == 0) {
261 speedSteps = atoi(_xml.data);
262 }
263 } else if (r == YXML_ELEMEND && strcmp(_xml.elem, "event") == 0) {
264 _listener->onLocoSpeedChanged(loco, speedPercent, direction, speedSteps);
265 }
266 }
267 yxml_eof(&_xml);
268 }
269
270private:
271 IUnifiedModelTrainListener* _listener;
272 yxml_t _xml;
273 uint8_t _stack[4096];

277public:
278 CmdLineParser(IUnifiedModelTrainListener& listener) : _listener(&listener) {}

280 String getParamValue(const String& params, const char* key) {
281 String searchKey = String(key) + "=\"";
282 int keyIndex = params.indexOf(searchKey);
283 if (keyIndex == -1) {
284 return "";
285 }
286 int valueStartIndex = keyIndex + searchKey.length();
287 int valueEndIndex = params.indexOf('"', valueStartIndex);
288 if (valueEndIndex == -1) {
289 return "";
290 }
291 return params.substring(valueStartIndex, valueEndIndex);
292 }
293
294 void parse(const String& command) {
295 if (command.length() == 0) {
296 return;
297 }
298
299 // Try parsing extended format first
300 int firstSpace = command.indexOf(' ');
301 String cmd;
302 String params;
303 if (firstSpace != -1) {
304 cmd = command.substring(0, firstSpace);
305 params = command.substring(firstSpace + 1);
306 } else {
307 cmd = command;
308 }
309
310 if (cmd == "POWER") {
311 String state = getParamValue(params, "state");
312 if (state == "ON") {
313 _listener->onTrackPowerChanged(PowerState::ON);
314 } else if (state == "OFF") {
315 _listener->onTrackPowerChanged(PowerState::OFF);
316 }
317 return;
318 } else if (cmd == "THROTTLE") {
319 uint16_t cab = getParamValue(params, "cab").toInt();
320 int speed = getParamValue(params, "speed").toInt();
321 int dir = getParamValue(params, "direction").toInt();
322 int steps = getParamValue(params, "steps").toInt();
323 if (steps == 0) steps = 128; // default
324
325 LocoHandle loco = {cab, Protocol::DCC, 0};
326 _listener->onLocoSpeedChanged(loco, speed, (Direction)dir, steps);
327 return;
328 } else if (cmd == "FUNCTION") {
329 uint16_t cab = getParamValue(params, "cab").toInt();
330 int func = getParamValue(params, "function").toInt();
331 int state = getParamValue(params, "state").toInt();
332
333 LocoHandle loco = {cab, Protocol::DCC, 0};
334 _listener->onLocoFunctionChanged(loco, func, state);
335 return;
336 } else if (cmd == "TURNOUT") {
337 uint16_t addr = getParamValue(params, "id").toInt();
338 int state = getParamValue(params, "state").toInt();
339 _listener->onTurnoutChanged(addr, state, false);
340 return;
341 } else if (cmd == "SIGNAL") {
342 uint16_t addr = getParamValue(params, "id").toInt();
343 int aspect = getParamValue(params, "aspect").toInt();
344 _listener->onSignalAspectChanged(addr, aspect, false);
345 return;
346 } else if (cmd == "MECH_SYNC") {
347 uint16_t cab = getParamValue(params, "cab").toInt();
348 String typeStr = getParamValue(params, "type");
349 uint8_t value = getParamValue(params, "value").toInt();
350 SyncType type;
351 if (typeStr == "CAM_PULSE") type = SyncType::CAM_PULSE;
352 else if (typeStr == "CYLINDER_CYCLE") type = SyncType::CYLINDER_CYCLE;
353 else if (typeStr == "GEAR_CHANGE_UP") type = SyncType::GEAR_CHANGE_UP;
354 else if (typeStr == "GEAR_CHANGE_DOWN") type = SyncType::GEAR_CHANGE_DOWN;
355 else if (typeStr == "BRAKE_SQUEAL_START") type = SyncType::BRAKE_SQUEAL_START;
356 else if (typeStr == "DOOR_MOVEMENT") type = SyncType::DOOR_MOVEMENT;
357
358 LocoHandle loco = {cab, Protocol::DCC, 0};
359 _listener->onMechanicalSyncEvent(loco, type, value);
360 return;
361 }
362
363 // If not an extended command, try parsing legacy format
364 char cmdChar = command.charAt(0);
365 params = command.substring(1);
366 params.trim();
367
368 switch (cmdChar) {
369 case '1':
370 _listener->onTrackPowerChanged(PowerState::ON);
371 break;
372 case '0':
373 _listener->onTrackPowerChanged(PowerState::OFF);
374 break;
375 case 't':
376 {
377 int firstSpace = params.indexOf(' ');
378 int secondSpace = params.indexOf(' ', firstSpace + 1);
379 if (firstSpace != -1 && secondSpace != -1) {
380 uint16_t cab = params.substring(0, firstSpace).toInt();
381 int speed = params.substring(firstSpace + 1, secondSpace).toInt();
382 int dir = params.substring(secondSpace + 1).toInt();
383
384 LocoHandle loco = {cab, Protocol::DCC, 0};
385 _listener->onLocoSpeedChanged(loco, speed, (Direction)dir, 128);
386 }
387 }
388 break;
389 case 'f':
390 {
391 int firstSpace = params.indexOf(' ');
392 int secondSpace = params.indexOf(' ', firstSpace + 1);
393 if (firstSpace != -1 && secondSpace != -1) {
394 uint16_t cab = params.substring(0, firstSpace).toInt();
395 int func = params.substring(firstSpace + 1, secondSpace).toInt();
396 int state = params.substring(secondSpace + 1).toInt();
397
398 LocoHandle loco = {cab, Protocol::DCC, 0};
399 _listener->onLocoFunctionChanged(loco, func, state);
400 }
401 }
402 break;
403 case 'T':
404 {
405 int firstSpace = params.indexOf(' ');
406 if (firstSpace != -1) {
407 uint16_t addr = params.substring(0, firstSpace).toInt();
408 int state = params.substring(firstSpace + 1).toInt();
409 _listener->onTurnoutChanged(addr, state, false);
410 }
411 }
412 break;
413 case 'S':
414 {
415 int firstSpace = params.indexOf(' ');
416 if (firstSpace != -1) {
417 uint16_t addr = params.substring(0, firstSpace).toInt();
418 int aspect = params.substring(firstSpace + 1).toInt();
419 _listener->onSignalAspectChanged(addr, aspect, false);
420 }
421 }
422 break;
423 case 'm':
424 {
425 int firstSpace = params.indexOf(' ');
426 int secondSpace = params.indexOf(' ', firstSpace + 1);
427 if (firstSpace != -1 && secondSpace != -1) {
428 uint16_t cab = params.substring(0, firstSpace).toInt();
429 int type = params.substring(firstSpace + 1, secondSpace).toInt();
430 int value = params.substring(secondSpace + 1).toInt();
431
432 LocoHandle loco = {cab, Protocol::DCC, 0};
433 _listener->onMechanicalSyncEvent(loco, (SyncType)type, value);
434 }
435 }
436 break;
437 }
438 }
